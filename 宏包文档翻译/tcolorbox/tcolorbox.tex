 {\begin{list}
  {}
  {\renewcommand\makelabel[1]{\hfil##1}
   \settowidth\labelwidth{\makelabel{#1}}
   \itemsep=-5pt%
   \setlength\leftmargin{%
   \labelwidth+\labelsep}}
