\begin{docKey}
  {--value, -v}{}%
  {}%
  Show value of command instead (i.e. \cs{the}\cs{command}).

  显示命令的值，而不是命令本身（即 \cs{the}\cs{command}）。

\begin{commandshell} 
latexdef -c '[a4paper]{book}' -v paperwidth
\end{commandshell}  

\begin{commandshell} 
latexdef -c '[letter]{book}'  -v paperwidth
\end{commandshell}  
\end{docKey}
